0001# Used to validate the user submitted form
0002# The schema is required in order to determine the requirements
0003
0004import formencode
0005
0006class UserForm(formencode.Schema):
0007
0008        # Other fields that don't require validation, such as select option
0009        allow_extra_fields = True
0010        filter_extra_fields = True
0011        # The username field should not be empty
0012        name = formencode.validators.NotEmpty()
0013        # The email field should not be empty and should contain @ sign
0014        email = formencode.validators.Email()
0015        # The password length should be at least 6 chars long
0016        password = formencode.validators.MinLength(6)
0017        # The birthday field should not be empty
0018        birthday = formencode.validators.NotEmpty()
0019        # The wage field should numeric
0020        wage = formencode.validators.Int()
