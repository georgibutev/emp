0001from sqlalchemy import Column
0002from sqlalchemy.types import Integer, String
0003
0004from emp.model.meta import Base
0005
0006class User(Base):
0007        # Mapper used for SQLalchemy operations
0008
0009        # The name of the table
0010        __tablename__ = "user"
0011
0012        # Table description
0013        # Auto incrementing numeric key to identify the emplyees.
0014        id = Column(Integer, primary_key=True)
0015        name = Column(String(33))
0016        email = Column(String(33))
0017        password = Column(String(25))
0018        birthday = Column(String(33))
0019        # Numeric value
0020        wage = Column(Integer)
0021        department = Column(String(33))
0022
0023        def __init__(self, name, email, password, birthday, wage, department):
0024        # Initialize the variables so that they are available in the class
0025                self.name = name
0026                self.email = email
0027                self.password = password
0028                self.birthday = birthday
0029                self.wage = wage
0030                self.department = department
0031
0032
0033        def __repr__(self):
0034        # Returs a string and digit that describes the User object
0035                return "<User('%s', '%s', '%s', '%s', %d, '%s')>" % (self.name, self.email, self.password, self.birthday, self.wage, self.department)
