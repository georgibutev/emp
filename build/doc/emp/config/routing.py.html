0001"""Routes configuration
0002
0003The more specific and detailed routes should be defined first so they
0004may take precedent over the more generic routes. For more information
0005refer to the routes manual at http://routes.groovie.org/docs/
0006"""
0007from routes import Mapper
0008
0009def make_map(config):
0010    """Create, configure and return the routes Mapper"""
0011    map = Mapper(directory=config['pylons.paths']['controllers'],
0012                 always_scan=config['debug'])
0013    map.minimization = False
0014    map.explicit = False
0015
0016    # The ErrorController route (handles 404/500 error pages); it should
0017    # likely stay at the top, ensuring it can always be resolved
0018    map.connect('/error/{action}', controller='error')
0019    map.connect('/error/{action}/{id}', controller='error')
0020
0021    # CUSTOM ROUTES HERE
0022    # Root directory route to main action
0023    map.connect('/', controller='app', action='main')
0024    map.connect('/{controller}/{action}')
0025    map.connect('/{controller}/{action}/{id}')
0026
0027    return map
